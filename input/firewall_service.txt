config firewall service custom
    edit "fortimanager"
        set category "Remote Access"
        set tcp-portrange 541
    next
    edit "fortiadmin"
        set tcp-portrange 8443
    next
end

config firewall service group
    edit "guest-services"
        set member "HTTP" "HTTPS" "IMAP" "IMAPS" "POP3" "POP3S" "SMTP" "SMTPS" "FTP" "PING" "DNS"
    next
end
